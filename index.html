import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import schedule
import time

def send_automated_email(recipient_email, recipient_name, wellness_score, template_number):
    """
    Send automated email based on assessment results
    """
    sender_email = "your-email@gmail.com"
    sender_password = "your-app-password"  # Use App Password, not regular password
    
    msg = MIMEMultipart()
    msg['From'] = sender_email
    msg['To'] = recipient_email
    
    # Select template based on template_number and wellness_score
    if template_number == 1:
        subject = f"Your Wellness Assessment Results - {recipient_name}'s Transformation Plan Inside! 🌟"
        body = generate_results_email(recipient_name, wellness_score)
    elif template_number == 2:
        subject = f"{recipient_name}, your transformation timeline is waiting... ⏰"
        body = generate_followup_email(recipient_name, wellness_score)
    # Add more templates as needed
    
    msg['Subject'] = subject
    msg.attach(MIMEText(body, 'html'))
    
    try:
        server = smtplib.SMTP('smtp.gmail.com', 587)
        server.starttls()
        server.login(sender_email, sender_password)
        text = msg.as_string()
        server.sendmail(sender_email, recipient_email, text)
        server.quit()
        print(f"Email sent successfully to {recipient_email}")
    except Exception as e:
        print(f"Error sending email: {str(e)}")

# Schedule emails
def schedule_email_sequence(recipient_email, recipient_name, wellness_score):
    # Immediate email
    send_automated_email(recipient_email, recipient_name, wellness_score, 1)
    
    # Schedule 24-hour follow-up
    schedule.every(24).hours.do(send_automated_email, recipient_email, recipient_name, wellness_score, 2)
    
    # Schedule 72-hour follow-up  
    schedule.every(72).hours.do(send_automated_email, recipient_email, recipient_name, wellness_score, 3)

# Run scheduler
while True:
    schedule.run_pending()
    time.sleep(3600)  # Check every hour
